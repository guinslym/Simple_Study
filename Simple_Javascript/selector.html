	function $(){
		var elems=[];
		for (var i = arguments.length-1; i >=0; i--) {
			var elem=arguments[i];
			if (typeof elem =='string') {
				elem=document.querySelector(elem);
			}
			if (arguments.length==1) {
				return elem;
			}
			elems.push(elem);
		}
		return elems;
	}
